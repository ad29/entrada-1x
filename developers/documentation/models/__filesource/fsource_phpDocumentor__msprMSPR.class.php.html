  285.     public function getMSPRRevisions($type="pdf", $location = null) {

  288.         }
  289.         $user = $this->getUser();
  290.         $search_string = $location . "/" .$user->getNumber()."-*.".$type; 
  291.         $files = glob($search_string);
  292.         
  293.         //extract timestamps - the only part we care about
  294.         $revisions = array();
  295.         foreach ($files as $file) {
  296.             $basename = basename($file,".".$type);
  297.             $parts = explode("-",$basename);
  298.             $revisions[] = $parts[1];
  299.         }
  300.         //sort by timestamp (newest first)
  301.         sort($revisions,SORT_NUMERIC);
  302.         return array_reverse($revisions);
  303.     }
