  37.     public function loadFile($filename) {
  38.         $filename = realpath($filename);
  39.         if (file_exists($filename)) {
  40.             $this->_template_data = new DOMDocument();
  41.             $this->_template_data->preserveWhiteSpace = false;
  42.             $this->_template_data->load($filename);
  43.             //note the above instruction to NOT preserve whitespace is required. PHP does not handle cdata sections properly if there is *any* text around them. this could obviously mess up a <pre> section if you have one

  45.         } else {

  50.     /**

  54.     public function loadString($str) {

  64.     public function createRestorePoint() {

  68.     /**
  69.      * Restores the template to a previously defined restore point. numerical index. negative values OK

  73.     public function restore($point = null) {
  74.         $rp_len = count($this->_restore_points);
  75.         if ($rp_len == 0) { //no restore points
  76.             return false;
  77.         }
  78.         if (is_int($point)) {
  79.             if ($point < 0) {
  80.                 $point =  $rp_len + $point; 
  81.             }
  82.             if ($point >= $rp_len) {
  83.                 return false;
  84.             }
  85.         } else {
  86.             $point = $rp_len-1;
  87.         }
  88.         $this->_template_data = $this->_restore_points[$point]->cloneNode(true); 
  89.         return true;
  90.     }

  92.     /**
  93.      * Alias for restore(0);

  97.     public function reset() {

  99.         $this->_restore_points = array( $this->_restore_points[0] );
  100.         return $success;
  101.     }

  103.     /**
  104.      * Applies the bound variables to the selected template
  105.      * @param array $bind_array (optional)
  106.      * @param array $select (optional)
  107.      */
  108.     public function apply(array $bind_array = array(), array $select = array()){
  109.         $working_template = $this->getTemplate($select);
  110.             
  111.         //now get every descendent and process with bound values
  112.         $xpath = new DOMXpath($this->_template_data);
  113.         $n_query = "./descendant-or-self::text()";
  114.         $nodes = $xpath->query($n_query, $working_template);
  115.         if ($nodes->length > 0) {
  116.             for($i=0,$len = $nodes->length; $i<$len;$i++) {
  117.                 $node = $nodes->item($i);
  118.                 $text = $node->wholeText;
  119.                 $text_len = strlen($text);
  120.                 $text = substitute_vars($text, $bind_array);
  121.                 $node->replaceData(0,$text_len,$text);
  122.             }
  123.         }
  124.     }

  126.     /**
  127.      * Retrieves the selected template
  128.      * @param array $select (optional)
  129.      * @return DOMNode 
  130.      */
  131.     private function getTemplate(array $select = array()) {
  132.         $selector ="";
  133.         if (count($select > 0)) {
  134.             reset($select);
  135.             $selector = "[";
  136.             do {
  137.                 $selector .= "@".key($select)."='".current($select)."'";
  138.             } while(next($select) && ($selector .= " and "));
  139.             $selector .="]";
  140.         }
  141.         
  142.         if ($this->_template_data) {
  143.             $xpath = new DOMXpath($this->_template_data);
  144.             $t_query = "//template".$selector;
  145.             
  146.             $t_data = $xpath->query($t_query);
  147.         
  148.             //only take the first result, if any
  149.             if ($t_data->length > 0) {
  150.                 return $t_data->item(0);
  151.             }
  152.         } else {
  153.             throw new Exception("Template not loaded.");
  154.         }
  155.     }

  157.     /**
  158.      * Returns the result and restores the template to the last restore point. optionally accepts bind values
  159.      * @param array $select (optional)
  160.      * @param array $bind_array (optional)
  161.      */
  162.     public function getResult(array $bind_array = array(), array $select = array()) {
  163.         $working_template = $this->getTemplate($select);
  164.         if ($working_template) {
  165.             $this->apply($bind_array, $select); //NOTE: this modifies $working_template!  
  166.             $sxl = simplexml_import_dom($working_template);
  167.             $this->restore(); //undo the changes to the template for the next getResult()
  168.             return $sxl;
  169.         }
